The javascript call

This simple UI merely consumes a REST endpoint opened up by Scailable. The core javascript to do so looks like this: 


// Create input and output objects; feature_vector should be a numerical vector
sclbl.input = {
    "content-type": "json",
    "location": "embedded",
    "data": "{\"input\": [" + JSON.stringify(feature_vector) + "]}"
};

sclbl.output = {
    "content-type": "json",
    "location": "echo"
};

// Endpoint settings
let endpointUrl = "https://taskmanager.sclbl.net:8080/
	task/45017963-8536-11ea-9efc-9600004e79cc";
		
let myHeaders = new Headers();
myHeaders.append("Content-Type", "application/x-www-form-urlencoded");

let raw = JSON.stringify(
    {
        input: sclbl.input,
        output: sclbl.output,
        control: 1,
        properties: {
            language: "WASM",
            profit: 0.5,
            cycles: 1
        }
    }
);

let requestOptions = {
    method: 'POST',
    headers: myHeaders,
    body: raw,
    redirect: 'follow'
};

// Do the actual call to the Scailable REST endpoint:
return fetch(endpointUrl, requestOptions)
	.then(response => {
	    if (!response.ok) {
	        throw response
	    }
	    return response.text()
	})
	.then(text => {
	    return (text)
	})
	.catch(err => {
	    return ("Error")
	})
